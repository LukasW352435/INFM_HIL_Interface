400 template<class Archive>
401 inline void
402 save_construct_data(Archive &ar, const sim_interface::dut_connector::can::CANConnectorReceiveOperation *config,
403 const unsigned int version) {
404
405 ar & boost::serialization::make_nvp("operation", config->operation);
406 ar & boost::serialization::make_nvp("isCANFD", config->isCANFD);
407 ar & boost::serialization::make_nvp("hasMask", config->hasMask);
408
409 std::stringstream hex;
410 hex << "0x";
411 hex << std::hex;
412 int _maskLength = CAN_MAX_DLEN;
413
414 if (config->isCANFD) {
415 _maskLength = CANFD_MAX_DLEN;
416 }
417 for (int i = 0; i < _maskLength; i++) {
418
419 hex << std::setw(2) << std::setfill('0') << (int) config->mask.data[i];
420 }
421
422 std::string stringHexValue = hex.str();
423 ar & boost::serialization::make_nvp("mask", stringHexValue);
424
425 }

439 template<class Archive>
440 inline void
441 load_construct_data(Archive &ar, sim_interface::dut_connector::can::CANConnectorReceiveOperation *instance,
442 const unsigned int file_version) {
443
444 std::string _operation;
445 bool _isCANFD;
446 bool _hasMask;
447 int _maskLength = CAN_MAX_DLEN;
448
449 std::string hexMask;
450
451 ar & boost::serialization::make_nvp("operation", _operation);
452 ar & boost::serialization::make_nvp("isCANFD", _isCANFD);
453 ar & boost::serialization::make_nvp("hasMask", _hasMask);
454 ar & boost::serialization::make_nvp("mask", hexMask);
455
456 __u8 _maskCANLength[CAN_MAX_DLEN] = {0};
457 __u8 _maskCANFDLength[CANFD_MAX_DLEN] = {0};
458 __u8 *_mask = _maskCANLength;
459
460 if (_isCANFD) {
461 _maskLength = CANFD_MAX_DLEN;
462 _mask = _maskCANFDLength;
463 }
464 // Parse every two chars into one byte
465 for (unsigned int i = 0; i < _maskLength && i < hexMask.length() - 2; i += 2) {
466 std::string byteString = hexMask.substr(i + 2, 2);
467 __u8 byte = (__u8) strtol(byteString.c_str(), NULL, 16);
468 _mask[i / 2] = byte;
469 }
470
471 ::new(instance)sim_interface::dut_connector::can::CANConnectorReceiveOperation(_operation, _isCANFD,
472 _hasMask, _maskLength, _mask
473 );
474 }
